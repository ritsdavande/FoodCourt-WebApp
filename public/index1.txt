import React, { useState, useEffect } from "react";

// Add a default value for 'options' to prevent the error
export default function Card({ foodName, ImgSrc, options = {} }) {
  const [qty, setQty] = useState(1);
  const [size, setSize] = useState("");
  const [finalPrice, setFinalPrice] = useState(0);

  const priceOptions = Object.keys(options);

  const handleQty = (e) => {
    setQty(e.target.value);
  };

  const handleSize = (e) => {
    setSize(e.target.value);
  };

  useEffect(() => {
    let price = 0;
    if (size) {
      price = qty * parseInt(options[size]);
    }
    setFinalPrice(price);
  }, [qty, size, options]);

  return (
    <div>
      <div className="card mt-3" style={{ width: "16rem", maxHeight: "360px" }}>
        <img
          src="https://images.unsplash.com/photo-1544025162-d76694265947?w=900&h=500&fit=crop"
          className="card-img-top"
          alt="..."
          style={{ height: "120px", objectFit: "fill" }}
        />
        <div className="card-body">
          <h5 className="card-title">{foodName}</h5>
          <div className="container w-100 p-0" style={{ height: "38px" }}>
            <select
              className="m-2 h-100 w-20 bg-success text-black rounded"
              onChange={handleQty}
            >
              {Array.from(Array(6), (e, i) => (
                <option key={i + 1} value={i + 1}>
                  {i + 1}
                </option>
              ))}
            </select>
            <select
              className="m-2 h-100 w-20 bg-success text-black rounded"
              onChange={handleSize}
            >
              <option value="">Select Size</option>
              {priceOptions.map((i) => (
                <option key={i} value={i}>
                  {i}
                </option>
              ))}
            </select>
            <div className="d-inline ms-2 h-100 w-20 fs-5">₹{finalPrice}/-</div>
          </div>
        </div>
      </div>
    </div>
  );
}
